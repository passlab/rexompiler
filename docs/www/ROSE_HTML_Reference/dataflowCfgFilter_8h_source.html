36  case V_SgCommaOpExp:

43

49  //purposes
50  case V_SgConditionalExp:
51  return (cfgn.getIndex() == 1);
52 
53  //Make these binary operators appear after their operands, because they are evaluated after their operands
54  case V_SgAndOp:
55  case V_SgOrOp:
